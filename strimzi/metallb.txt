helm repo add metallb https://metallb.github.io/metallb
helm install metallb metallb/metallb

kubectl apply -f - << EOF
apiVersion: metallb.io/v1beta1
kind: IPAddressPool
metadata:
  name: demo-pool
  namespace: default
spec:
  addresses:
  - 172.18.0.100-172.18.0.140
EOF



kubectl apply -f - << EOF
apiVersion: metallb.io/v1beta1
kind: L2Advertisement
metadata:
  name: example
  namespace: default
spec:
  ipAddressPools:
  - demo-pool
EOF
